function read32bitInt(data,startIndex){
  var _id = 0;
  var shifter = 24;
  for(var a = startIndex; a < startIndex+4; a++){
    _id |= (data[a] << shifter);
    shifter -= 8;
  }
  return _id;
}
 
 
function write32BitInt(buffer,offset,address){
  for(var a = 0; a < 4; a++){
